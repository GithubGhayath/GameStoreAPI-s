CREATE DATABASE GameStore;

USE GameStore

CREATE TABLE Games(
	Id INT PRIMARY KEY IDENTITY(1,1),
	Name NVARCHAR(MAX) NOT NULL,
	Price SMALLMONEY NOT NULL,
	ReleasDate DATE NOT NULL
);


CREATE TABLE Geners(
	Id INT PRIMARY KEY IDENTITY(1,1),
	Name NVARCHAR(MAX) NOT NULL
);
